```cpp
// Program that uses break statement

#include <iostream>
using namespace std;

int main() {
    for (int i = 1; i <= 10; i++) {
        if(i==4) {
            break;
        }
            cout << i << endl;
    }
}
```

```cpp
//Program that uses continue statement

#include <iostream>
using namespace std;

int main() {
    for (int i = 1; i <= 10; i++) {
        if(i==4) {
            continue;
        }
        cout << i << endl;
    }
    return 0;
}
```
